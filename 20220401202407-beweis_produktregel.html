\begin{align*} g'(x) \coloneqq& \lim_{\varepsilon \to 0}\left(\frac{g(x + \varepsilon) - g(x)}{\varepsilon}\right) \\ h'(x) \coloneqq& \lim_{\varepsilon \to 0}\left(\frac{h(x + \varepsilon) - h(x)}{\varepsilon}\right) \end{align*} \begin{align*} f'(x) =& \lim_{\varepsilon \to 0}\left(\frac{f(x + \varepsilon) - f(x)}{\varepsilon}\right) \\ =& \lim_{\varepsilon \to 0}\left(\frac{g(x+\varepsilon) \cdot h(x+\varepsilon) - g(x) \cdot h(x)}{\varepsilon }\right) \\ =& \lim_{\varepsilon \to 0}\left(\frac{g(x+\varepsilon) \cdot h(x+\varepsilon) - g(x) \cdot h(x) + g(x)h(x+\varepsilon) - g(x)h(x+\varepsilon)}{\varepsilon }\right) \\ =& \lim_{\varepsilon \to 0}\left(\frac{h(x+\varepsilon) \cdot \left(g(x+\varepsilon) - g(x)\right) + g(x)\cdot\left(h(x+\varepsilon) - h(x)\right)}{\varepsilon }\right) \\ =& \lim_{\varepsilon \to 0}\left(h(x+\varepsilon\right)) \cdot \lim_{\varepsilon \to 0}\left(\frac{g(x+\varepsilon) - g(x)}{\varepsilon}\right) + \lim_{\varepsilon \to 0}\left(g(x\right)) \lim_{\varepsilon \to 0}\left(\frac{h(x+\varepsilon) - h(x)}{\varepsilon}\right) \\ =& h(x) \cdot g'(x) + g(x) \cdot h'(x) \end{align*}
